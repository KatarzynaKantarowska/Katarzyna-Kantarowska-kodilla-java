SELECT SUMMARY, DESCRIPTION, NAME
FROM ISSUES I JOIN ISSUESLISTS IL on  I.ID = IL.ID;

SELECT SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U on I.ID = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(ISSUESLIST_ID)
FROM ISSUES I JOIN USERS U on I.ID = U.ID
GROUP BY U.ID, U.FIRSTNAME, U.LASTNAME;
